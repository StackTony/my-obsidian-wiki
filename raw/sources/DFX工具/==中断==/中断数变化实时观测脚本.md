
该脚本用于动态观测和统计系统的每秒中断数，显示约1s内增加的中断数量及其对应的CPU。支持设定每秒中断数屏蔽阈值，只显示大于阈值的数据。脚本不依赖其他模块，便于直接以文本方式复制到虚拟机内运行。
```
#!/bin/bash

function Clean()
{
	[[ -e pre ]] && rm -f pre
	[[ -e cur ]] && rm -f cur
	exit 0
}
trap Clean SIGHUP SIGINT SIGTERM

th=1   # 中断数量屏蔽阈值
CPUS=`lscpu | grep '^CPU(s):' | awk '{print $NF}'`
space="                                                                    "
while true; do
	echo -en "\033[1;33mIgnore delta <= ${th} / s\033[0m\n\n"
	cat /proc/interrupts | grep -E ':([[:space:]]*[0-9]+){3,}' | grep -v arch_timer > pre
	sleep 1
	cat /proc/interrupts | grep -E ':([[:space:]]*[0-9]+){3,}' | grep -v arch_timer > cur
	a=`diff pre cur | grep '^<' | cut -c3-`
	b=`diff pre cur | grep '^>' | cut -c3-`
	readarray -t a_rows <<<"$a"
	readarray -t b_rows <<<"$b"
	for ((i=0; i<${#a_rows[@]}; i++)); do
		IFS=' ' read -r -a a_values <<< "${a_rows[i]}"
		IFS=' ' read -r -a b_values <<< "${b_rows[i]}"
		num=0
		for ((j=0; j<${#a_values[@]}; j++)); do
			if [ ${a_values[j]} != ${b_values[j]} ]; then
				delta=$((${b_values[j]}-${a_values[j]}))
				if [ ${delta} -gt ${th} ]; then
					intID=$(printf "%-6s" ${a_values[0]})
					num=${#intID}
					echo -en "\033[1;32m${intID}\033[0m"
					for ((j=$((CPUS+1)); j<${#a_values[@]}; j++)); do
						echo -en " ${a_values[j]}"
						num=$((${num}+${#a_values[j]}+1))
					done
					echo -en "\t\t"
					break
				fi
			fi
		done
		prefix="${space:0:$num}\t\t"
		fillSpace=""
		for ((j=0; j<${#a_values[@]}; j++)); do
			if [ ${a_values[j]} != ${b_values[j]} ]; then
				delta=$((${b_values[j]}-${a_values[j]}))
				if [ ${delta} -gt ${th} ]; then
					line=$(printf "${fillSpace}cpu%4s: %10s / s     %10s -> %10s" $((j-1)) ${delta} ${a_values[j]} ${b_values[j]})
					echo -en "$line\n"
					[[ "${fillSpace}" == "" ]] && fillSpace="${prefix}"
				fi
			fi
		done
	done
done
```
